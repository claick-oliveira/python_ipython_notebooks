def is_float(number):
    try:
        float(number)
        return True
    except:
        return False


def fix_blanks_and_nans(col):
    col = col.fillna('Missing_Value')
    col[col == ''] = 'Blank_Value'
    return col


data_frame = pd.read_csv('my_file.csv', sep=',')
for col in data_frame.columns:
    data_frame[col] = fix_blanks_and_nans(data_frame[col])
